# ###########################
# Add a new remote to GitHub
# ###########################
git remote add origin https://github.com/geridantech/GIT.git
# Set a new remote
git remote -v
# Verify new remote
origin  https://github.com/geridantech/GIT.git (fetch)
origin  https://github.com/geridantech/GIT.git (push)

# Rename a Remote Branch
git remote rename <old> <new>

#Undo Init (delete directory)
rm -rf .git

#View staged changes
git diff --cached

#Add existing repository 
git remote add Chocolatey https://github.com/geridantech/Chocolatey.git
git push -u Chocolatey master
